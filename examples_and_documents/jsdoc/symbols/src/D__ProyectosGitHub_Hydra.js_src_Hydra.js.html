727 		 * Checks the completion of all the Promise objects.
728 		 * If any of the Promises is not complete continue waiting.
729 		 * If all the Promises are complete then complete method is executed.
730 		 * @member Deferred.prototype
731 		 */
732 		checkCompleted: function () {
733 			var nPromise, nLenPromise, oPromise;
734 			nLenPromise = this.aPromises.length;
735 			oPromise = _null_;
736 
737 			for (nPromise = 0; nPromise < nLenPromise; nPromise++) {
738 				oPromise = this.aPromises[nPromise];
739 				if (!oPromise.bCompleted) {
740 					return _false_;
741 				}
742 			}
743 			this.complete(this.getType());
744 
745 			nPromise = _null_;
746 			nLenPromise = _null_;
747 			oPromise = _null_;
748 		},
749 		/**
750 		 * Return the type of completion [reject, resolve]
751 		 * @member Deferred.prototype
752 		 */
753 		getType: function () {
754 			var nPromise, nLenPromises, oPromise, aTypes, sType;
755 			nLenPromises = this.aPromises.length;
756 			oPromise = _null_;
757 			aTypes = [];
758 
759 			for (nPromise = 0; nPromise < nLenPromises; nPromise++) {
760 				oPromise = this.aPromises[nPromise];
761 				if (oPromise.bCompleted) {
762 					aTypes.push(oPromise.sType);
763 				}
764 			}
765 			if (aTypes.length === 0) {
766 				return "";
767 			}
768 			sType = aTypes.join("").replace(reResolve, "").length > 0 ? 'reject' : 'resolve';
769 
770 			try {
771 				return sType;
772 			} finally {
773 				nPromise = _null_;
774 				nLenPromises = _null_;
775 				oPromise = _null_;
776 				aTypes = _null_;
777 			}
778 		},
779 		/**
780 		 * Executes the correct method/s depending of completion type [success, fail]
781 		 * @member Deferred.prototype
782 		 * @param {String} sType
783 		 */
784 		complete: function (sType) {
785 			var nPromise, nLenPromises, oPromise;
786 			if (sType.length === 0) {
787 				return _false_;
788 			}
789 			nLenPromises = this.aPromises.length;
790 
791 			for (nPromise = 0; nPromise < nLenPromises; nPromise++) {

794 					oPromise.aPending.shift()[sType](oPromise.oResult);
795 				}
796 			}
797 			while (this.aPending[0]) {
798 				this.aPending.shift()[sType]();
799 			}
800 
801 			nPromise = _null_;
802 			nLenPromises = _null_;
803 			oPromise = _null_;
804 		},

808 		 * @param {Function} fpFailure

810 		then: function (fpSuccess, fpFailure) {
811 			this.aPending.push({ resolve: fpSuccess, reject: fpFailure});
812 			return this;
813 		}
