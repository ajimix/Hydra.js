 32 	/**
 33 	 * wrapMethod is a method to wrap the original method to avoid failing code.
 34 	 * This will be only called if bDebug flag is setted to false.
 35 	 * @private
 36 	 * @author Tomas Corral Casas
 37 	 * @version 1.0
 38 	 * @param {Object} oInstance
 39 	 * @param {String} sName
 40 	 * @param {String} sModuleId
 41 	 * @param {Function} fpMethod
 42 	 */
 43 	function wrapMethod(oInstance, sName, sModuleId, fpMethod) {
 44 		oInstance[sName] = (function (sName, fpMethod) {
 45 			return function () {
 46 				var aArgs = Array.prototype.slice.call(arguments, 0);
 47 				try {
 48 					return fpMethod.apply(this, aArgs);
 49 				} catch (erError) {
 50 					ErrorHandler.log(sModuleId, sName, erError);
 51 				} finally {
 52 					aArgs = null;
 53 				}
 54 			};
 55 		}(sName, fpMethod));
 56 	}
 57 	/**
 58 	 * createInstance is the method that will create the module instance and wrap the method if needed.
 59 	 * @private
 60 	 * @author Tomas Corral Casas
 61 	 * @version 1.0
 62 	 * @param {String} sModuleId
 63 	 * @return {Object} Module instance
 64 	 */
 65 	function createInstance(sModuleId) {
 66 		if(typeof oModules[sModuleId] === "undefined"){
 67 			throw new Error("The module is not registered!");
 68 		}
 69 		var oInstance = oModules[sModuleId].creator(new Action()),
 70 			sName = '',
 71 			fpMethod = function () {};
 72 
 73 		if (!bDebug) {
 74 			for (sName in oInstance) {
 75 				if (oInstance.hasOwnProperty(sName)) {
 76 					fpMethod = oInstance[sName];
 77 					if (!isFunction(fpMethod)) {
 78 						continue;
 79 					}
 80 					wrapMethod(oInstance, sName, sModuleId, fpMethod);
 81 				}
 82 			}
 83 		}
 84 		return oInstance;
 85 	}

155 	/**
156 	 * log is the method that will differenciate the system if they had console or not.
157 	 * if windowdow.console exist console.log will be called
158 	 * if windowdow.console not exist then the log on layer will be activated.
159 	 * Lazy pattern will be used to avoid extra work on next calls.
160 	 * Arguments are sent to the methods that will be applied.
161 	 * @static
162 	 * @author Tomas Corral Casas
163 	 * @version 1.0
164 	 */
165 	ErrorHandler.log = function () {
166 		var aArgs = Array.prototype.slice.call(arguments, 0);
167 
168 		if (typeof windowdow.console === "undefined" || (typeof aArgs[aArgs.length  - 1] == 'boolean' && !aArgs[aArgs.length - 1])) {
169 			if (this.list === null) {
170 				this.create_dom();
171 			}
172 			this.addItem.apply(this, aArgs);
173 			ErrorHandler.__old_log__ = ErrorHandler.log;
174 			this.log = function () {
175 				var aArgs = Array.prototype.slice.call(arguments, 0);
176 				this.addItem.apply(this, aArgs);
177 				aArgs = null;
178 			};
179 		} else {
180 			window.console.log.apply(window.console, aArgs);
181 			ErrorHandler.__old_log__ = ErrorHandler.log;
182 			this.log = function () {
183 				var aArgs = Array.prototype.slice.call(arguments, 0);
184 				window.console.log.apply(window.console, aArgs);
185 				aArgs = null;
186 			};
187 		}
188 		aArgs = null;
189 	};

218 	 * @member Action.prototype
219 	 * @author Tomas Corral Casas
220 	 * @version 1.0
221 	 * @param {Array} aNotificationsToListen
222 	 * @param {Function} fpHandler
223 	 * @param {Object} oModule
224 	 */
225 	Action.prototype.listen = function (aNotificationsToListen, fpHandler, oModule) {
226 		var sNotification = '',
227 			nNotification = 0,
228 			nLenNotificationsToListen = aNotificationsToListen.length;
229 
230 		for (; nNotification < nLenNotificationsToListen; nNotification++) {
231 			sNotification = aNotificationsToListen[nNotification];
232 			if (typeof Action.oActions[sNotification] === "undefined") {
233 				Action.oActions[sNotification] = [];
234 			}
235 			Action.oActions[sNotification].push({
236 				module: oModule,
237 				handler: fpHandler
238 			});
239 		}

242 	 * notify is the method that will launch the actions that are listening the notified action
243 	 * @member Action.prototype
244 	 * @author Tomas Corral Casas
245 	 * @version 1.0
246 	 * @param oNotifier - Notifier.type and Notifier.data are needed
247 	 */
248 	Action.prototype.notify = function (oNotifier) {
249 		var sType = oNotifier.type,
250 			oAction = null,
251 			nAction = 0,
252 			nLenActions = 0,
253 			oActions = Action.oActions;
254 
255 		if (typeof oActions[sType] === "undefined") {
256 			return;
257 		}
258 
259 		nLenActions = oActions[sType].length;
260 		for (; nAction < nLenActions; nAction++) {
261 			oAction = oActions[sType][nAction];
262 			oAction.handler.call(oAction.module, oNotifier);
263 		}
264 		sType = oAction = nAction = nLenActions = oActions = null;
265 	};

271 	 * @param {Array} aNotificationsToStopListen
272 	 * @param {Object} oModule
273 	 */
274 	Action.prototype.stopListen = function (aNotificationsToStopListen, oModule) {
275 		var sNotification = '',
276 			aAuxActions = [],
277 			nNotification = 0,
278 			nLenNotificationsToListen = aNotificationsToStopListen.length,
279 			nAction = 0,
280 			nLenActions = 0,
281 			oActions = Action.oActions;
282 
283 		for (; nNotification < nLenNotificationsToListen; nNotification++) {
284 			sNotification = aNotificationsToStopListen[nNotification];
285 			nLenActions = oActions[sNotification].length;
286

288 				if (oModule !== oActions[sNotification][nAction].module) {

290 				}
291 			}
292 			oActions[sNotification] = aAuxActions;
293 			if (oActions[sNotification].length === 0) {
294 				delete oActions[sNotification];
295 			}
296 		}

342 	 * @member Module.prototype
343 	 * @private
344 	 * @author Tomas Corral Casas
345 	 * @version 1.0
346 	 * @param {Object} oModuleBase
347 	 * @param {Object} oModuleExtended
348 	 * @return {Object} the merged module
349 	 */
350 	Module.prototype.merge = function (oModuleBase, oModuleExtended) {
351 		var oFinalModule = {},
352 			sKey = '';
353 
354 		for (sKey in oModuleBase) {
355 			if (oModuleBase.hasOwnProperty(sKey)) {
356 				oFinalModule[sKey] = oModuleBase[sKey];
357 			}
358 		}
359 
360 		for (sKey in oModuleExtended) {
361 			if (oModuleExtended.hasOwnProperty(sKey)) {
362 				oFinalModule[sKey] = oModuleExtended[sKey];
363 			}
364 		}
365 		return oFinalModule;
366 	};

372 	 * @version 1.0
373 	 * @param {String} sModuleId
374 	 * @param {Function} fpCreator
375 	 */
376 	Module.prototype.extend = function (sModuleId, fpCreator) {
377 		var oModule = oModules[sModuleId],
378 			self = this,
379 			oBaseModule = null,
380 			oExtendedModule = null,
381 			oFinalModule = null;
382

384 			return;
385 		}
386 
387 		oExtendedModule = fpCreator(new Action());
388 		oBaseModule = oModule.creator(new Action());
389 		oFinalModule = this.merge(oBaseModule, oExtendedModule);
390

392 			creator: function (oAction) {
393 				return oFinalModule;
394 			},
395 			instance: null
396 		};
397 
398 		oModule = oBaseModule = oExtendedModule = null;
399 	};
